### 快速上手

#### 下载

``` js
    npm i -D choas-vue
    cnpm i -D choas-vue
    yarn add -D choas-vue
```


#### 使用

```js
import Vue from 'vue'
import App from './App.vue'
import ChoasUI from './lib'
Vue.config.productionTip = false
Vue.use(ChoasUI)
new Vue({
  render: h => h(App),
}).$mount('#app')

```

```js
<template>
  <div id="app">
    <Button></Button>
  </div>
</template>
```
